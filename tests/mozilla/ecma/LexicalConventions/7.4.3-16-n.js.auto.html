    startTest();
    var TITLE   = "Future Reserved Words";

    writeHeaderToLog( SECTION + " "+ TITLE);

    var testcases = new Array();

    var result = "Failed";
    var exception = "No exception thrown";
    var expect = "Passed";

    try {
        try = true;
    } catch ( e ) {
        result = expect;
        exception = e.toString();
    }

    testcases[tc++] = new TestCase(
        SECTION,
        "try = true" +
        " (threw " + exception +")",
        expect,
        result );

    test();
